\frametitle{FastQ Example}

\begin{python}
def mean(xs):
    return sum(xs)/float(len(xs))
class FastQSequence(object):
    def __init__(self, seq, quals):
        if len(seq) != len(quals):
            print 'OOOOOOOOOOPS!'
        self.seq = seq
        self.quals = quals

    def averageq(self):
        return mean(self.quals)
\end{python}

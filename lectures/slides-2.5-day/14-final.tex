\begin{frame}[fragile]
\frametitle{Examples of HTSeq Usage}

\begin{python}
maxq = 0
for seq in HTSeq.FastqReader('hw-HeLa.fq.gz', 'solexa'):
    maxq = max(maxq, seq.qual.max())

print 'Maximum quality found was {0}.'.format(maxq)
\end{python}
